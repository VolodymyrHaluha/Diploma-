-- Database schema for the ZenithFit site
-- Target DBMS: PostgreSQL

DROP TABLE IF EXISTS attendance;
DROP TABLE IF EXISTS subscriptions;
DROP TABLE IF EXISTS equipment_usage;
DROP TABLE IF EXISTS class_sessions;
DROP TABLE IF EXISTS news_posts;
DROP TABLE IF EXISTS trainers;
DROP TABLE IF EXISTS clients;
DROP TABLE IF EXISTS membership_plans;
DROP TABLE IF EXISTS users;

-- Function for automatic updated_at update
CREATE OR REPLACE FUNCTION set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;


CREATE TABLE membership_plans (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    price NUMERIC(10, 2) NOT NULL,
    period_days INTEGER NOT NULL DEFAULT 30,
    description TEXT,
    created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT uq_membership_plans_name UNIQUE (name),
    CONSTRAINT chk_membership_plans_price CHECK (price >= 0),
    CONSTRAINT chk_membership_plans_period_days CHECK (period_days > 0)
);


CREATE TABLE clients (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    full_name VARCHAR(150) NOT NULL,
    phone VARCHAR(30),
    email VARCHAR(150) NOT NULL,
    photo_url VARCHAR(500),
    status VARCHAR(20) NOT NULL DEFAULT 'active',
    created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT uq_clients_email UNIQUE (email),
    CONSTRAINT chk_clients_status CHECK (status IN ('active', 'inactive', 'blocked'))
);


CREATE TABLE trainers (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    full_name VARCHAR(150) NOT NULL,
    specialty VARCHAR(150) NOT NULL,
    experience VARCHAR(100),
    photo_url VARCHAR(500),
    created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE class_sessions (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    trainer_id INTEGER NOT NULL,
    title VARCHAR(150) NOT NULL,
    day_of_week VARCHAR(20) NOT NULL,
    start_time TIME NOT NULL,
    capacity INTEGER NOT NULL,
    created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_class_sessions_trainer
        FOREIGN KEY (trainer_id) REFERENCES trainers(id)
        ON UPDATE CASCADE ON DELETE RESTRICT,

    CONSTRAINT chk_class_sessions_day_of_week
        CHECK (day_of_week IN (
            'Monday',
            'Tuesday',
            'Wednesday',
            'Thursday',
            'Friday',
            'Saturday',
            'Sunday'
        )),

    CONSTRAINT chk_class_sessions_capacity CHECK (capacity > 0)
);


CREATE TABLE subscriptions (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    client_id INTEGER NOT NULL,
    plan_id INTEGER NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_subscriptions_client
        FOREIGN KEY (client_id) REFERENCES clients(id)
        ON UPDATE CASCADE ON DELETE CASCADE,

    CONSTRAINT fk_subscriptions_plan
        FOREIGN KEY (plan_id) REFERENCES membership_plans(id)
        ON UPDATE CASCADE ON DELETE RESTRICT,

    CONSTRAINT chk_subscriptions_dates CHECK (end_date >= start_date)
);


CREATE TABLE attendance (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    client_id INTEGER NOT NULL,
    class_session_id INTEGER NOT NULL,
    visit_time TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    check_type VARCHAR(20) NOT NULL DEFAULT 'check_in',

    CONSTRAINT fk_attendance_client
        FOREIGN KEY (client_id) REFERENCES clients(id)
        ON UPDATE CASCADE ON DELETE CASCADE,

    CONSTRAINT fk_attendance_class_session
        FOREIGN KEY (class_session_id) REFERENCES class_sessions(id)
        ON UPDATE CASCADE ON DELETE CASCADE,

    CONSTRAINT chk_attendance_check_type
        CHECK (check_type IN ('check_in', 'check_out'))
);


CREATE TABLE equipment_usage (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    equipment_name VARCHAR(150) NOT NULL,
    usage_value NUMERIC(10, 2) NOT NULL,
    measured_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT chk_equipment_usage_value CHECK (usage_value >= 0)
);


CREATE TABLE users (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    username VARCHAR(80) NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role VARCHAR(20) NOT NULL DEFAULT 'editor',
    created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT uq_users_username UNIQUE (username),
    CONSTRAINT chk_users_role CHECK (role IN ('admin', 'editor', 'trainer'))
);


CREATE TABLE news_posts (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    author_id INTEGER NOT NULL,
    title VARCHAR(200) NOT NULL,
    tag VARCHAR(80),
    content TEXT NOT NULL,
    created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_news_posts_author
        FOREIGN KEY (author_id) REFERENCES users(id)
        ON UPDATE CASCADE ON DELETE RESTRICT
);


-- Indexes

CREATE INDEX idx_clients_status
    ON clients (status);

CREATE INDEX idx_trainers_specialty
    ON trainers (specialty);

CREATE INDEX idx_class_sessions_day_time
    ON class_sessions (day_of_week, start_time);

CREATE INDEX idx_class_sessions_trainer_id
    ON class_sessions (trainer_id);

CREATE INDEX idx_subscriptions_client_id
    ON subscriptions (client_id);

CREATE INDEX idx_subscriptions_plan_id
    ON subscriptions (plan_id);

CREATE INDEX idx_subscriptions_active_dates
    ON subscriptions (is_active, start_date, end_date);

CREATE INDEX idx_attendance_client_id
    ON attendance (client_id);

CREATE INDEX idx_attendance_class_session_id
    ON attendance (class_session_id);

CREATE INDEX idx_attendance_visit_time
    ON attendance (visit_time);

CREATE INDEX idx_equipment_usage_name
    ON equipment_usage (equipment_name);

CREATE INDEX idx_equipment_usage_measured_at
    ON equipment_usage (measured_at);

CREATE INDEX idx_users_role
    ON users (role);

CREATE INDEX idx_news_posts_author_id
    ON news_posts (author_id);

CREATE INDEX idx_news_posts_tag
    ON news_posts (tag);

CREATE INDEX idx_news_posts_created_at
    ON news_posts (created_at);


-- Triggers for updated_at

CREATE TRIGGER trg_membership_plans_updated_at
BEFORE UPDATE ON membership_plans
FOR EACH ROW
EXECUTE FUNCTION set_updated_at();

CREATE TRIGGER trg_clients_updated_at
BEFORE UPDATE ON clients
FOR EACH ROW
EXECUTE FUNCTION set_updated_at();

CREATE TRIGGER trg_trainers_updated_at
BEFORE UPDATE ON trainers
FOR EACH ROW
EXECUTE FUNCTION set_updated_at();

CREATE TRIGGER trg_class_sessions_updated_at
BEFORE UPDATE ON class_sessions
FOR EACH ROW
EXECUTE FUNCTION set_updated_at();

CREATE TRIGGER trg_subscriptions_updated_at
BEFORE UPDATE ON subscriptions
FOR EACH ROW
EXECUTE FUNCTION set_updated_at();

CREATE TRIGGER trg_users_updated_at
BEFORE UPDATE ON users
FOR EACH ROW
EXECUTE FUNCTION set_updated_at();

CREATE TRIGGER trg_news_posts_updated_at
BEFORE UPDATE ON news_posts
FOR EACH ROW
EXECUTE FUNCTION set_updated_at();
